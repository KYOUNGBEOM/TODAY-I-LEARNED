간단한 권장 칼로리 계산기 만들기 - 자바스크립트만을 사용하였음.

let sex = prompt("성별은 입력하세요", "남자 또는 여자");

if (sex === "남자") {
    let age = prompt("나이를 입력하세요", "예: 20");
    let weight = prompt("몸무게를 입력하세요","예: 60");
    let height = prompt("키를 입력하세요", "예: 172");
    result = (66.47 + (13.75 * weight) + (5 * height) - (6.76 * age));
} else if (sex === "여자") {
    let age = prompt("나이를 입력하세요", "예: 20");
    let weight = prompt("몸무게를 입력하세요","예: 60");
    let height = prompt("키를 입력하세요", "예: 172");
    result = (665.1 + (9.56 * weight) + (1.85 * height) - (4.68 * age));
} else {
    alert("다시 입력해주세요.")
}

let calorie = Math.round(result)

document.write(`하루 권장 ${calorie} kcal`);
